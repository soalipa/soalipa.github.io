        = 169 - 13
        = 156
     t = √156
       = 2 √39 cm

b. tΔ² = 13² - 3²
         = 169 - 9
         = 160 
    tΔ₁ = √160 
          = 4√10  cm
 
    tΔ² = 13² - 2²
          = 169 - 4
          = 165 
    tΔ₂ = √165 cm

Lp = La + 2LΔ₁ + 2LΔ₂
     =(6 x 4) + 2(1/2 x 6 x 4√10) + 2(1/2 x 4 x √165)
     = (24 + 24√10 + 4√165) cm²

c. LN = KM = √52
                  = 2√13 cm
  
    L = 1/2 x a x t
       = 1/2 x 2√13 x 2√39
       = 1/2 x 4√507
       = 1/2 x 4 x 14√3
       = 28√3 cm²

d. V = 1/3 x La x tl
       = 1/3 x (6 x 4) x 2√39
       = 1/3 x 24 x 2√39
       = 16√39  cm³
