a. x² + 6x + 5 = 0

x² + 6x = -5

x² + 6x + (6/2)² = -5 + (6/2)²

x² + 6x + 3² = -5 + 36/4

(x + 3)(x + 3) = -20/4 + 36/4

(x +3)² = 16/4

(x + 3)² = 4

(x + 3) = √4

x + 3 = ± 2

Terdapat dua penyelesaian :

. x + 3 = + 2

x = -1

. x + 3 = -2

x = -5

b. 2x² - 4x - 6 = 0

bagi 2

x² - 2x - 3 = 0

x² - 2x = 3

x² - 2x + (-2/2)² = 3 + (-2/2)²

x² - 2x + 1 = 3 + 1

x² - 2x + 1² = 4

(x - 1)² = 4

x - 1 = √4

x - 1 = ± 2

Terdapat dua penyelesaian :

. x - 1 = + 2

x = + 2 + 1

x = 3

. x - 1 = -2
